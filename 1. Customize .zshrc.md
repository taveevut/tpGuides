php_use() {
  local v="$1"  # เช่น 7.4 หรือ 8.2
  # ตัด php เดิมออกจาก PATH
  PATH="$(echo "$PATH" | tr ':' '\n' | awk '!/\/php@/ {print}' | paste -sd: -)"

  export PATH="/usr/local/opt/php@${v}/bin:/usr/local/opt/php@${v}/sbin:$PATH"
  php -v
}
